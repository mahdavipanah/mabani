          int factorial (int number) {
            int fact_number = 1;
            for (; number > 1; number--)
              fact_number *= number;
            return fact_number;
          }

            int i, temp;
            for (i = 0; i < len / 2; i++) {
              temp = ar[i];
              ar[i] = ar[len - i - 1];
              ar[len - i - 1] = temp;
            }
          }

            // if ( 97 <= letter && letter <= 122)
            if ('a' <= letter && letter <= 'z')
              return 1;
            return 0;
          }

          int is_upper_case(const char letter) {
            // if ( 65 <= letter && letter <= 90)
            if ('A' <= letter && letter <= 'Z')
              return 1;
            return 0;
          }

            if (is_upper_case(letter))
              // return letter + ('a' - 'A');
              return letter + 32;
            return letter;
          }

          // Defined in above examples
          int is_lower_case(const char);

          char to_upper_case(const char letter) {
            if (is_lower_case(letter))
              // return letter - ('a' - 'A');
              return letter - 32;
            return letter;
          }

          int count_char(const char * str, const char c) {
              int char_counts = 0;
              for (int index = 0; str[index]; index++)
                  if (str[index] == c)
                      char_counts++;
              return char_counts;
          }

            int remove_counts = 0;
            for (int index = first_char_index(str, c); index != -1;
                 index = first_char_index(str, c), remove_counts++)

        }

  // order = 0 for Ascending and order = 1 for Descending
  void array_sort(int * a, int len, int order) {
    for (int i = 0; i < len - 1; i++)
      for (int j = i + 1; j < len; j++)
            if ((a[i] > a[j] && !order) || (a[i] < a[j] && order) ) {
              int temp = a[i];
              a[i] = a[j];
              a[j] = temp;
            }
  }

          #include <math.h>
          #include <string.h>

          // String must contain numeric characters (1...9)
          unsigned int str_to_uint(const char * str) {
              unsigned int ret_val = 0;
              for (int i = 0, j = strlen(str) - 1; str[i];
                  i++, j--)
                  ret_val += pow(10, j) * (str[i] - 48);
              return ret_val;
          }
